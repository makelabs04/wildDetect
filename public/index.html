ESP32 Arduino Sketch
Flash this to your ESP32 — Green = clear, Red blink = animal detected

📦 Requires: ArduinoJson (Library Manager)
📌 Pins: Green LED→26, Red LED→27, Buzzer→25 (set -1 if none)
🌐 Edit SSID, PASSWORD before flashing 
// ╔══════════════════════════════════════════════════════╗
// ║   WildDetect ESP32 — LED Indicator                   ║
// ║   Green SOLID = No animal detected                   ║
// ║   Red BLINK   = Animal / Bird detected!              ║
// ╚══════════════════════════════════════════════════════╝
#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>   // Library Manager → ArduinoJson by Bblanchon

const char* SSID       = "YOUR_WIFI_SSID";
const char* PASSWORD   = "YOUR_WIFI_PASSWORD";
const char* SERVER_URL = "https://wildeye.makelearners.com/api/led";

#define PIN_GREEN   26
#define PIN_RED     27
#define PIN_BUZZER  25    // set to -1 if no buzzer
#define POLL_INTERVAL_MS  1000

enum LedMode { MODE_GREEN, MODE_RED };
LedMode currentMode = MODE_GREEN;
unsigned long blinkNextMs = 0;
bool blinkOn = false;
const unsigned long BLINK_HALF_MS = 250;

struct BuzzTask { int left=0; bool on=false; unsigned long nextMs=0; } buzz;
String lastLedCode = "";
unsigned long lastPollMs = 0;

void allOff() {
  digitalWrite(PIN_GREEN,LOW); digitalWrite(PIN_RED,LOW);
  if(PIN_BUZZER>=0) digitalWrite(PIN_BUZZER,LOW);
}
void applyGreen() { currentMode=MODE_GREEN; allOff(); digitalWrite(PIN_GREEN,HIGH); }
void applyRed()   { currentMode=MODE_RED; blinkOn=true; blinkNextMs=millis(); allOff(); digitalWrite(PIN_RED,HIGH); }

void setup() {
  Serial.begin(115200);
  pinMode(PIN_GREEN,OUTPUT); pinMode(PIN_RED,OUTPUT);
  if(PIN_BUZZER>=0) pinMode(PIN_BUZZER,OUTPUT);
  allOff();
  WiFi.begin(SSID, PASSWORD);
  Serial.print("WiFi connecting");
  unsigned long t=millis();
  while(WiFi.status()!=WL_CONNECTED){
    delay(300); Serial.print(".");
    digitalWrite(PIN_GREEN,!digitalRead(PIN_GREEN));
    if(millis()-t>20000){ ESP.restart(); }
  }
  allOff();
  for(int i=0;i<3;i++){ digitalWrite(PIN_GREEN,HIGH);digitalWrite(PIN_RED,HIGH);delay(120);allOff();delay(120); }
  applyGreen();
  Serial.println("\n[OK] Ready");
}

void loop() {
  if(millis()-lastPollMs>=POLL_INTERVAL_MS){ lastPollMs=millis(); pollServer(); }
  runLed(); runBuzz();
}

void pollServer() {
  if(WiFi.status()!=WL_CONNECTED){ WiFi.reconnect(); return; }
  HTTPClient http; http.begin(SERVER_URL); http.setTimeout(3000);
  int code=http.GET();
  if(code!=200){ http.end(); return; }
  String body=http.getString(); http.end();
  StaticJsonDocument<256> doc;
  if(deserializeJson(doc,body)||!doc["ok"]) return;
  String ledCode=doc["data"]["led_code"]|"G";
  String label=doc["data"]["label"]|"";
  int buzzCount=doc["data"]["buzz"]|0;
  if(ledCode==lastLedCode) return;
  lastLedCode=ledCode;
  if(ledCode=="G"){ applyGreen(); Serial.println("[LED] GREEN"); }
  else if(ledCode=="R"){
    applyRed(); Serial.println("[LED] RED BLINK — "+label);
    if(buzzCount>0&&PIN_BUZZER>=0){ buzz.left=buzzCount; buzz.on=false; buzz.nextMs=millis()+100; }
  }
}

void runLed() {
  if(currentMode==MODE_GREEN) return;
  unsigned long now=millis();
  if(now>=blinkNextMs){ blinkOn=!blinkOn; digitalWrite(PIN_RED,blinkOn?HIGH:LOW); blinkNextMs=now+BLINK_HALF_MS; }
}

void runBuzz() {
  if(buzz.left<=0||PIN_BUZZER<0) return;
  unsigned long now=millis(); if(now<buzz.nextMs) return;
  if(!buzz.on){ digitalWrite(PIN_BUZZER,HIGH); buzz.on=true; buzz.nextMs=now+130; }
  else{ digitalWrite(PIN_BUZZER,LOW); buzz.on=false; buzz.left--; buzz.nextMs=now+200; }
}
